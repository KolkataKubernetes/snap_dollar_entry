\begin{frame}{Research Question}
\alert{\textbf{Motivation.}}
  \begin{itemize}
     \item Many assistance programs aim to improve access to essential goods through demand-side subsidies.
     \item These policies typically assume market structure will remain the same.
     \item However, large enough demand shifts can generate strategic supply responses.
  \end{itemize}

  \alert{\textbf{Entry responses are ambiguous.}}
  \begin{itemize}
     \item New entrants can lower prices downwards and increase variety.
     \item New entrants are not necessarily offering what the policy targets. 
  \end{itemize}

  \alert{\textbf{Questions.}}
  \begin{itemize}
     \item Do retailers' entry behavior respond to SNAP expansions?
     \item Do these responses ultimately affect SNAP intended policy outcome (food security)?
  \end{itemize}
\end{frame}



\begin{frame}{Literature Review}
This project bridges three strands of literature:
\begin{itemize}
    \item \alert{\textbf{SNAP and store choice.}} Benefit increases have modest direct effects on where recipients shop. A \$100 increase shifts $\sim$1pp of spending toward grocery stores \citep{cuffey2022effects}.
    \item \alert{\textbf{Retailer composition and consumption.}} But the \emph{type} of retailer matters: dollar store entry reduces produce consumption by 7--30\%, concentrated among low-income households \citep{caoui2025dollar}.
    \item \alert{\textbf{Capture of consumer aid.}} Work in the education literature has shown that financial aid for higher education has been captured by for-profit schools \citep{Cellini2010FinancialAidForProfit,CelliniGoldin2014FederalAidTuition}
\end{itemize}
\alert{This paper:} Do SNAP expansions induce retailer entry? If so, does the composition of entrants reinforce or undermine the policy's food security goals?
\end{frame}

\begin{frame}{Background: SNAP}
\begin{itemize}
    \item SNAP provides monthly transfers (\$177/person on average) redeemable at authorized food retailers.
    \item Largest U.S. food assistance program: $\sim$42 million participants, $\sim$\$100 billion annually.
    \item Eligibility based on income, assets, and household composition.
    \item For able-bodied adults without dependents (ABAWDs), benefits are time-limited: 3 months in a 36-month period unless work requirements are met.
\end{itemize}
\end{frame}

\begin{frame}{Source of Variation: ABAWD Waivers}
\begin{itemize}
    \item States can request \alert{waivers} of the ABAWD time limit for areas with:
    \begin{itemize}
        \item Unemployment $>$10\%, or
        \item Insufficient jobs to provide employment
    \end{itemize}
    \item Waivers expand SNAP by removing the 3-month limit for this population.
    \item WHY IS IT RELEVANT: SHOW $\%$ INCREASE FROM PAPERS. GRAY PAPER MAY BE A STARTING POINT
    \item \alert{Key variation:} Waivers granted at the {\alert{county level}} and change over time with local labor market conditions.

\end{itemize}
\end{frame}

\begin{frame}{Reduced form Model}
\begin{equation}
Y_{it} = \sum_{k=-3, k \neq 0}^{4} \beta_k D_{it}^k + \mathbf{X}_{it}'\gamma + \alpha_i + \lambda_t + \varepsilon_{it}
\end{equation}

Where:
\begin{itemize}
    \item $Y_{it}$ is the inverse hyperbolic sine (IHS) transformation of stores in county $i$ at time $t$
    \item $D_{it}^k$ are event-time indicators equal to 1 when county $i$ is $k$ periods from treatment
    \item $\mathbf{X}_{it}$ is a vector of time-varying controls including population, average wage, mean income, rent, and unemployment rate
    \item $\alpha_i$ are county fixed effects
    \item $\lambda_t$ are year fixed effects
    \item $\varepsilon_{it}$ is the error term
\end{itemize}
\end{frame}
